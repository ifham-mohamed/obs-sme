> **Goal:** deliver Module 2 end-to-end — three SME survey instruments (Awareness, Compliance Knowledge Test, Vulnerability), the scoring engine that fuses them into a 0–100 knowledge score, the RAG knowledge base built from Module 1 regulation summaries, and the inter-module contract that feeds Module 3.

## 3. Question-Bank Loader

YAML is authored by the research lead and reviewed by the CA. The loader is idempotent on `question_code` and refuses to insert any item without `verified_at`.

```python
# FILE: backend/app/modules/m2/seed_questions.py
import asyncio
from pathlib import Path
import yaml
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.db.session import SessionLocal
from app.models.m2 import M2Question

QUESTION_DIR = Path("data/m2/questions")

REQUIRED_FOR_PROD = ("ground_truth_source", "ground_truth_verified_by", "verified_at")

async def upsert_question(db: AsyncSession, item: dict) -> None:
    if item["instrument"] == "knowledge_test":
        for f in REQUIRED_FOR_PROD:
            if not item.get(f):
                raise ValueError(f"{item['question_code']}: missing CA verification field {f}")
    existing = (await db.execute(
        select(M2Question).where(M2Question.question_code == item["question_code"])
    )).scalar_one_or_none()
    if existing:
        for k, v in item.items():
            setattr(existing, k, v)
    else:
        db.add(M2Question(**item))

async def main() -> None:
    async with SessionLocal() as db:
        for path in sorted(QUESTION_DIR.glob("*.yaml")):
            for item in yaml.safe_load(path.read_text(encoding="utf-8")):
                await upsert_question(db, item)
        await db.commit()
        print("seeded m2_questions")

if __name__ == "__main__":
    asyncio.run(main())
```

CLI: `python -m app.modules.m2.seed_questions`. Run after every CA-approved revision; the loader never silently downgrades a verified item.

```python
# FILE: backend/app/api/v1/m2_surveys.py
from uuid import UUID, uuid4
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.api.deps import get_db, current_sme
from app.models.m2 import M2Question, SurveyResponse

router = APIRouter(prefix="/api/v1/m2/surveys", tags=["m2"])

class StartReq(BaseModel):
    instrument: str

class StartResp(BaseModel):
    resume_token: UUID
    questions: list[dict]

@router.post("/start", response_model=StartResp)
async def start(req: StartReq, db: AsyncSession = Depends(get_db), sme=Depends(current_sme)):
    if req.instrument not in {"awareness", "knowledge_test", "vulnerability"}:
        raise HTTPException(400, "unknown instrument")
    qs = (await db.execute(
        select(M2Question)
        .where(M2Question.instrument == req.instrument, M2Question.retired_at.is_(None))
        .order_by(M2Question.question_id)
    )).scalars().all()
    return StartResp(
        resume_token=uuid4(),
        questions=[{
            "question_id": q.question_id,
            "code": q.question_code,
            "type": q.question_type,
            "prompt": {"en": q.prompt_en, "si": q.prompt_si, "ta": q.prompt_ta},
            "options": q.options_json,
            "branching": q.branching_logic,
            "subscale": q.subscale,
        } for q in qs],
    )

class AutosaveItem(BaseModel):
    question_id: int
    raw_value: dict
    time_spent_ms: int | None = None

class AutosaveReq(BaseModel):
    instrument: str
    resume_token: UUID
    items: list[AutosaveItem]

@router.post("/autosave")
async def autosave(req: AutosaveReq, db: AsyncSession = Depends(get_db), sme=Depends(current_sme)):
    for item in req.items:
        db.add(SurveyResponse(
            sme_id=sme.sme_id, question_id=item.question_id,
            instrument=req.instrument, raw_value=item.raw_value,
            time_spent_ms=item.time_spent_ms, resume_token=req.resume_token,
        ))
    await db.commit()
    return {"saved": len(req.items)}
```

The `submit` endpoint (omitted) finalises an attempt, calls the scoring engine, and writes to `sme_knowledge_scores`.

        <Button onClick={() => setIdx((i) => Math.min(i + 1, visible.length - 1))}>{t("next")}</Button>

## 6. Scoring Engine

```python
# FILE: backend/app/modules/m2/scoring.py
from dataclasses import dataclass
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.models.m2 import M2Question, SurveyResponse, SmeKnowledgeScore

# Composite weights — defended in research/04 §6 (Likert weighted lower than tested knowledge).
COMPOSITE_WEIGHTS = {"awareness": 0.25, "knowledge_test": 0.50, "vulnerability": 0.25}

@dataclass
class InstrumentScore:
    instrument: str
    raw: float
    n_items: int
    score_0_100: float

def _score_item(q: M2Question, raw: dict) -> float:
    if q.question_type == "likert5":
        v = float(raw.get("likert") or raw.get("value", 0))
        v = (6 - v) if q.reverse_scored else v
        return (v - 1) / 4  # → 0..1
    if q.question_type in ("mcq_single", "binary"):
        correct = (q.correct_answer or {}).get("value")
        return 1.0 if raw.get("value") == correct else 0.0
    if q.question_type == "mcq_multi":
        correct = set((q.correct_answer or {}).get("set", []))
        chosen = set(raw.get("set", []))
        if not correct: return 0.0
        return len(correct & chosen) / len(correct | chosen)  # Jaccard
    return 0.0

async def score_instrument(db: AsyncSession, sme_id: int, instrument: str) -> InstrumentScore:
    rows = (await db.execute(
        select(SurveyResponse, M2Question)
        .join(M2Question, M2Question.question_id == SurveyResponse.question_id)
        .where(SurveyResponse.sme_id == sme_id, SurveyResponse.instrument == instrument)
    )).all()
    if not rows:
        return InstrumentScore(instrument, 0.0, 0, 0.0)
    weighted_sum = 0.0
    weight_total = 0.0
    for resp, q in rows:
        s = _score_item(q, resp.raw_value)
        weighted_sum += s * float(q.weight)
        weight_total += float(q.weight)
    raw = weighted_sum / weight_total if weight_total else 0.0
    return InstrumentScore(instrument, raw, len(rows), round(raw * 100, 2))

async def recompute_for_sme(db: AsyncSession, sme_id: int) -> SmeKnowledgeScore:
    aw = await score_instrument(db, sme_id, "awareness")
    kt = await score_instrument(db, sme_id, "knowledge_test")
    vu = await score_instrument(db, sme_id, "vulnerability")
    composite = (
        aw.score_0_100 * COMPOSITE_WEIGHTS["awareness"]
        + kt.score_0_100 * COMPOSITE_WEIGHTS["knowledge_test"]
        + vu.score_0_100 * COMPOSITE_WEIGHTS["vulnerability"]
    )
    row = await db.get(SmeKnowledgeScore, sme_id) or SmeKnowledgeScore(sme_id=sme_id)
    row.awareness_score = aw.score_0_100
    row.knowledge_test_score = kt.score_0_100
    row.vulnerability_score = vu.score_0_100
    row.composite_score = round(composite, 2)
    db.add(row); await db.commit(); await db.refresh(row)
    return row
```

`sector_percentile` is computed in a nightly job (`app/tasks/m2_recompute.py`) by ranking each `composite_score` within its `sme_profiles.sector_code` bucket.

## 7. Reliability + Validity

```python
# FILE: ml/m2/reliability.py
"""Cronbach α + KR-20 + test-retest. Run after each fielding wave."""
from __future__ import annotations
import numpy as np
import pandas as pd

def cronbach_alpha(item_matrix: np.ndarray) -> float:
    """item_matrix: (n_respondents, k_items). For Likert / continuous items."""
    k = item_matrix.shape[1]
    item_var = item_matrix.var(axis=0, ddof=1)
    total_var = item_matrix.sum(axis=1).var(ddof=1)
    return (k / (k - 1)) * (1 - item_var.sum() / total_var)

def kr20(binary_matrix: np.ndarray) -> float:
    """For 0/1 scored knowledge-test items."""
    k = binary_matrix.shape[1]
    p = binary_matrix.mean(axis=0)
    q = 1 - p
    total_var = binary_matrix.sum(axis=1).var(ddof=1)
    return (k / (k - 1)) * (1 - (p * q).sum() / total_var)

def test_retest(scores_t1: pd.Series, scores_t2: pd.Series) -> float:
    return scores_t1.corr(scores_t2, method="pearson")

def report(matrices: dict[str, np.ndarray]) -> pd.DataFrame:
    rows = []
    for subscale, m in matrices.items():
        is_binary = set(np.unique(m)).issubset({0, 1})
        rows.append({
            "subscale": subscale,
            "n_items": m.shape[1],
            "n_respondents": m.shape[0],
            "alpha_or_kr20": kr20(m) if is_binary else cronbach_alpha(m),
            "method": "KR-20" if is_binary else "Cronbach α",
        })
    return pd.DataFrame(rows)
```

**Acceptance target:** every subscale ≥ 0.7. Items below the bar are flagged for revision before the next fielding wave; the YAML loader will not promote them.

## 8. RAG Knowledge Base — Ingestion

The KB is built from Module 1's already-translated summaries (`regulations.summary_en/si/ta`). We do **not** re-extract or re-translate; we stand on M1's output. Embedding model is `intfloat/multilingual-e5-base` (env `EMBEDDING_MODEL`, set in BUILD_03).

```python
# FILE: ml/m2/build_kb.py
"""Chunk Module 1 regulation summaries, embed, persist to ChromaDB.

Idempotent on (regulation_id, language, chunk_index, text_hash). Re-running after
a new gazette ingest only adds the deltas.
"""
from __future__ import annotations
import asyncio, hashlib, os
from typing import Iterable
import chromadb
from chromadb.config import Settings
from langchain.text_splitter import RecursiveCharacterTextSplitter
from sentence_transformers import SentenceTransformer
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from tqdm import tqdm
from app.db.session import SessionLocal
from app.models.regulation import Regulation

EMBEDDING_MODEL = os.environ["EMBEDDING_MODEL"]    # intfloat/multilingual-e5-base
COLLECTION = "regulations_chunks_v1"

splitter = RecursiveCharacterTextSplitter(
    chunk_size=512, chunk_overlap=64,
    separators=["\n\n", "\n", ". ", " "],
)

def _hash(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def _chunks_for(reg: Regulation) -> Iterable[tuple[str, str, int, str]]:
    for lang in ("en", "si", "ta"):
        text = getattr(reg, f"summary_{lang}") or ""
        if len(text) < 40:
            continue
        for i, c in enumerate(splitter.split_text(text)):
            yield (lang, c, i, _hash(c))

async def main() -> None:
    client = chromadb.PersistentClient(path="./chroma", settings=Settings(anonymized_telemetry=False))
    coll = client.get_or_create_collection(name=COLLECTION, metadata={"hnsw:space": "cosine"})
    model = SentenceTransformer(EMBEDDING_MODEL)

    existing_ids = set(coll.get(include=[])["ids"])

    async with SessionLocal() as db:
        regs = (await db.execute(select(Regulation).order_by(Regulation.regulation_id))).scalars().all()

    pending: list[tuple[str, str, dict]] = []
    for reg in tqdm(regs, desc="regulations"):
        for lang, chunk, idx, h in _chunks_for(reg):
            doc_id = f"{reg.regulation_id}:{lang}:{idx}:{h[:12]}"
            if doc_id in existing_ids:
                continue
            pending.append((doc_id, chunk, {
                "regulation_id": reg.regulation_id,
                "gazette_date": str(reg.gazette_date),
                "gazette_number": reg.gazette_number,
                "language": lang,
                "category": reg.predicted_category,
                "text_hash": h,
            }))

    BATCH = 64
    for i in range(0, len(pending), BATCH):
        batch = pending[i:i + BATCH]
        # e5 expects "passage: " prefix on documents
        embs = model.encode([f"passage: {c[1]}" for c in batch], normalize_embeddings=True).tolist()
        coll.add(
            ids=[c[0] for c in batch],
            documents=[c[1] for c in batch],
            metadatas=[c[2] for c in batch],
            embeddings=embs,
        )
    print(f"upserted {len(pending)} chunks into {COLLECTION}")

if __name__ == "__main__":
    asyncio.run(main())
```

CLI: `python -m ml.m2.build_kb`. The job is wired into `BUILD_12` scheduler to run nightly after the gazette pipeline finishes.

## 9. RAG QA Endpoint

```python
# FILE: backend/app/modules/m2/rag.py
import os
from dataclasses import dataclass
from typing import Literal
import chromadb
from sentence_transformers import SentenceTransformer
from langchain_community.vectorstores import Chroma
from langchain.chains import RetrievalQA
from langchain_community.llms import Ollama  # swap for OpenAI in demo mode
from langchain.embeddings.base import Embeddings

EMBEDDING_MODEL = os.environ["EMBEDDING_MODEL"]
COLLECTION = "regulations_chunks_v1"

class E5Embeddings(Embeddings):
    def __init__(self) -> None:
        self.m = SentenceTransformer(EMBEDDING_MODEL)
    def embed_documents(self, texts: list[str]) -> list[list[float]]:
        return self.m.encode([f"passage: {t}" for t in texts], normalize_embeddings=True).tolist()
    def embed_query(self, text: str) -> list[float]:
        return self.m.encode([f"query: {text}"], normalize_embeddings=True)[0].tolist()

@dataclass
class Citation:
    regulation_id: int
    gazette_date: str
    snippet: str

@dataclass
class RagAnswer:
    answer: str
    citations: list[Citation]

_chain: RetrievalQA | None = None

def _get_chain() -> RetrievalQA:
    global _chain
    if _chain is not None:
        return _chain
    client = chromadb.PersistentClient(path="./chroma")
    vs = Chroma(client=client, collection_name=COLLECTION, embedding_function=E5Embeddings())
    retriever = vs.as_retriever(search_kwargs={"k": 5})
    llm = Ollama(model="llama3:8b-instruct")
    _chain = RetrievalQA.from_chain_type(
        llm=llm, retriever=retriever, return_source_documents=True, chain_type="stuff",
    )
    return _chain

async def ask(question: str, language: Literal["en", "si", "ta"] = "en") -> RagAnswer:
    chain = _get_chain()
    out = await chain.ainvoke({"query": question})
    cites = [
        Citation(
            regulation_id=int(d.metadata["regulation_id"]),
            gazette_date=d.metadata["gazette_date"],
            snippet=d.page_content[:240],
        )
        for d in out.get("source_documents", [])
    ]
    return RagAnswer(answer=out["result"], citations=cites)
```

## 10. RAGAS Evaluation

```python
# FILE: ml/m2/eval_ragas.py
"""Run the golden 50-Q set through the RAG pipeline and emit RAGAS metrics."""
from __future__ import annotations
import asyncio, json, sys
from pathlib import Path
import yaml
from datasets import Dataset
from ragas import evaluate
from ragas.metrics import answer_relevancy, faithfulness, context_precision, context_recall
from app.modules.m2.rag import ask, _get_chain

GOLDEN = Path("ml/m2/golden_set.yaml")  # 50 CA-vetted Q/A with expected citations

THRESHOLDS = {"faithfulness": 0.85, "answer_relevancy": 0.80, "context_precision": 0.75}

async def collect() -> Dataset:
    items = yaml.safe_load(GOLDEN.read_text())
    rows = []
    for it in items:
        ans = await ask(it["question"], it.get("language", "en"))
        # Pull contexts from the chain's last retrieval
        retriever = _get_chain().retriever
        ctxs = [d.page_content for d in await retriever.aget_relevant_documents(it["question"])]
        rows.append({
            "question": it["question"],
            "answer": ans.answer,
            "contexts": ctxs,
            "ground_truth": it["expected_answer"],
        })
    return Dataset.from_list(rows)

async def main() -> None:
    ds = await collect()
    res = evaluate(ds, metrics=[faithfulness, answer_relevancy, context_precision, context_recall])
    summary = {k: float(v) for k, v in res.items()}
    print(json.dumps(summary, indent=2))
    Path("ml/m2/last_eval.json").write_text(json.dumps(summary, indent=2))
    fails = [k for k, t in THRESHOLDS.items() if summary.get(k, 0) < t]
    if fails:
        print(f"FAIL: below threshold for {fails}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    asyncio.run(main())
```

The golden set lives in `ml/m2/golden_set.yaml` — 50 Q/A pairs derived from real gazettes between Jan 2025 and April 2026, each with the CA-confirmed answer and at least one expected `regulation_id` citation.
